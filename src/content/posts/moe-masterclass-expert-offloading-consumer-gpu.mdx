Mixtral 8x7B has 46.7B total parameters but only activates 12.9B per token. At FP16, 46.7B parameters require 93.4 GB of GPU memory — far beyond any consumer GPU. But at INT4, only 23.4 GB is needed, which fits on a 24GB RTX 4090. The problem: even at INT4, some configurations still need offloading, and FP16/INT8 always does. This post builds a complete expert offloading system from scratch.

## 1. The Memory Problem

### Parameter Distribution in Mixtral

```python
def mixtral_memory_breakdown():
    """
    Detailed memory breakdown for Mixtral 8x7B.
    """
    d_model = 4096
    d_ff = 14336
    num_layers = 32
    num_experts = 8
    num_kv_heads = 8
    num_heads = 32
    head_dim = d_model // num_heads  # 128
    vocab_size = 32000

    # Per-layer expert parameters (SwiGLU: w1, w2, w3)
    # w1: [d_model, d_ff], w2: [d_ff, d_model], w3: [d_model, d_ff]
    expert_params = 3 * d_model * d_ff  # 176M per expert
    all_experts_per_layer = expert_params * num_experts  # 1.41B per layer

    # Per-layer attention parameters
    # Q: [d_model, d_model], K: [d_model, kv_dim], V: [d_model, kv_dim], O: [d_model, d_model]
    kv_dim = num_kv_heads * head_dim  # 1024
    attn_params = (d_model * d_model +    # Q
                   d_model * kv_dim +      # K
                   d_model * kv_dim +      # V
                   d_model * d_model)       # O
    # ~25M per layer

    # Router parameters per layer
    router_params = d_model * num_experts  # 32K per layer

    # Embedding
    embed_params = vocab_size * d_model  # 131M

    # Norms (negligible)
    norm_params = d_model * 2 * num_layers  # ~260K total

    total = (all_experts_per_layer + attn_params + router_params) * num_layers + embed_params

    return {
        "expert_params_per_layer_M": expert_params / 1e6,
        "all_experts_per_layer_B": all_experts_per_layer / 1e9,
        "attention_per_layer_M": attn_params / 1e6,
        "total_expert_params_B": all_experts_per_layer * num_layers / 1e9,
        "total_params_B": total / 1e9,
        "expert_fraction": (all_experts_per_layer * num_layers) / total,
    }

# Result: ~88% of parameters are in expert FFNs
```

The critical insight: 88% of Mixtral's parameters are in expert FFNs, and only 2 of 8 experts activate per token per layer. At any given token, 75% of expert parameters are idle.

## 2. The Offloading Strategy

### Tiered Memory Hierarchy

The idea: keep only the currently active experts on GPU. Move inactive experts to CPU RAM, and if RAM is insufficient, to SSD. The hierarchy:

1. **GPU VRAM** (fastest, most limited): Active experts + attention layers + KV cache
2. **CPU RAM** (fast, moderate capacity): Inactive experts in pinned memory
3. **SSD/NVMe** (slow, large capacity): Overflow when CPU RAM is insufficient

```python
import torch
import threading
from collections import OrderedDict
from enum import Enum

class StorageTier(Enum):
    GPU = "gpu"
    CPU = "cpu"
    SSD = "ssd"

class ExpertLocation:
    def __init__(self, layer_idx, expert_idx, tier, data=None, path=None):
        self.layer_idx = layer_idx
        self.expert_idx = expert_idx
        self.tier = tier
        self.data = data        # CPU tensor or GPU tensor
        self.path = path        # SSD file path
        self.last_used = 0      # Step counter for LRU eviction
        self.use_count = 0      # Total usage count
        self.loading = False    # True if being loaded asynchronously
```

### Expert Offloading Manager

```python
class ExpertOffloadManager:
    """
    Manages expert parameter placement across GPU/CPU/SSD.
    Supports prefetching and LRU eviction.
    """
    def __init__(
        self,
        model,
        num_layers,
        num_experts,
        gpu_expert_budget,    # Max experts on GPU at once
        cpu_expert_budget,    # Max experts in CPU RAM
        ssd_cache_dir="/tmp/expert_cache",
        device="cuda:0",
    ):
        self.model = model
        self.num_layers = num_layers
        self.num_experts = num_experts
        self.gpu_budget = gpu_expert_budget
        self.cpu_budget = cpu_expert_budget
        self.ssd_dir = ssd_cache_dir
        self.device = device
        self.step = 0

        # Track location of every expert
        self.locations = {}
        for layer in range(num_layers):
            for expert in range(num_experts):
                self.locations[(layer, expert)] = ExpertLocation(
                    layer, expert, StorageTier.CPU
                )

        # GPU LRU cache
        self.gpu_cache = OrderedDict()  # (layer, expert) -> tensor
        self.gpu_count = 0

        # CPU pinned memory pool
        self.cpu_cache = OrderedDict()

        # CUDA streams for async transfers
        self.transfer_stream = torch.cuda.Stream(device=device)

        # Prefetch queue
        self.prefetch_queue = []

    def _get_expert_params(self, layer_idx, expert_idx):
        """Extract expert parameters as a flat tensor."""
        expert = self.model.layers[layer_idx].moe.experts[expert_idx]
        params = []
        for p in expert.parameters():
            params.append(p.data.reshape(-1))
        return torch.cat(params)

    def _set_expert_params(self, layer_idx, expert_idx, flat_tensor):
        """Write flat tensor back into expert parameters."""
        expert = self.model.layers[layer_idx].moe.experts[expert_idx]
        offset = 0
        for p in expert.parameters():
            numel = p.numel()
            p.data = flat_tensor[offset:offset + numel].reshape(p.shape)
            offset += numel

    def ensure_on_gpu(self, layer_idx, expert_idx):
        """
        Ensure an expert's parameters are on GPU.
        Returns the expert module ready for computation.
        """
        key = (layer_idx, expert_idx)
        self.step += 1

        if key in self.gpu_cache:
            # Already on GPU — update LRU
            self.gpu_cache.move_to_end(key)
            self.locations[key].last_used = self.step
            self.locations[key].use_count += 1
            return self.model.layers[layer_idx].moe.experts[expert_idx]

        # Need to load to GPU
        loc = self.locations[key]

        if loc.tier == StorageTier.CPU:
            data = loc.data
        elif loc.tier == StorageTier.SSD:
            data = torch.load(loc.path, map_location="cpu")
        else:
            raise ValueError(f"Expert {key} in unknown state")

        # Evict from GPU if at capacity
        if self.gpu_count >= self.gpu_budget:
            self._evict_gpu_lru()

        # Transfer to GPU
        gpu_data = data.to(self.device, non_blocking=True)
        self._set_expert_params(layer_idx, expert_idx, gpu_data)

        self.gpu_cache[key] = gpu_data
        self.gpu_count += 1
        self.locations[key].tier = StorageTier.GPU
        self.locations[key].last_used = self.step
        self.locations[key].use_count += 1

        return self.model.layers[layer_idx].moe.experts[expert_idx]

    def _evict_gpu_lru(self):
        """Evict least recently used expert from GPU to CPU."""
        if not self.gpu_cache:
            return

        # Pop oldest (least recently used)
        key, gpu_data = self.gpu_cache.popitem(last=False)
        layer_idx, expert_idx = key

        # Move to CPU pinned memory
        cpu_data = gpu_data.to("cpu", non_blocking=True).pin_memory()

        # Update tracking
        self.locations[key].tier = StorageTier.CPU
        self.locations[key].data = cpu_data
        self.cpu_cache[key] = cpu_data
        self.gpu_count -= 1

        # If CPU is over budget, evict to SSD
        if len(self.cpu_cache) > self.cpu_budget:
            self._evict_cpu_to_ssd()

    def _evict_cpu_to_ssd(self):
        """Evict least recently used CPU expert to SSD."""
        if not self.cpu_cache:
            return

        key, cpu_data = self.cpu_cache.popitem(last=False)
        layer_idx, expert_idx = key

        path = f"{self.ssd_dir}/expert_L{layer_idx}_E{expert_idx}.pt"
        torch.save(cpu_data, path)

        self.locations[key].tier = StorageTier.SSD
        self.locations[key].data = None
        self.locations[key].path = path
```

## 3. Prediction-Based Prefetching

### The Problem with Reactive Loading

Without prefetching, every expert load from CPU to GPU is on the critical path. A PCIe 4.0 x16 transfer at ~25 GB/s for a 176M parameter expert at FP16 (352 MB):

$$t_{\text{transfer}} = \frac{352 \text{ MB}}{25 \text{ GB/s}} = 14.1 \text{ ms}$$

With 32 layers and 2 experts per layer, the worst case (all cache misses) is:

$$t_{\text{worst}} = 64 \times 14.1 \text{ ms} = 902 \text{ ms per token}$$

This is unacceptable. Prefetching hides this latency by loading the next layer's experts while the current layer computes.

### Router-Based Prefetching

```python
class PrefetchingExpertManager(ExpertOffloadManager):
    """
    Extends ExpertOffloadManager with prediction-based prefetching.
    Uses the router logits from layer N to prefetch experts for layer N+1.
    """
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.prefetch_stream = torch.cuda.Stream(device=self.device)
        self.prefetch_events = {}

    def prefetch_for_next_layer(self, current_layer, router_logits, top_k=2):
        """
        After routing decisions for current_layer, predict which experts
        will be needed in the next layer and start loading them.
        """
        next_layer = current_layer + 1
        if next_layer >= self.num_layers:
            return

        # Strategy 1: Use current layer's top-k experts as a predictor
        # Assumption: expert choice is correlated across adjacent layers
        probs = torch.softmax(router_logits, dim=-1)
        top_experts = probs.mean(dim=0).topk(top_k).indices  # Most popular experts

        with torch.cuda.stream(self.prefetch_stream):
            for e in top_experts.tolist():
                key = (next_layer, e)
                if key not in self.gpu_cache:
                    self._async_load_to_gpu(next_layer, e)

    def _async_load_to_gpu(self, layer_idx, expert_idx):
        """Load expert to GPU asynchronously on prefetch stream."""
        key = (layer_idx, expert_idx)
        loc = self.locations[key]

        if loc.tier == StorageTier.GPU:
            return

        if loc.tier == StorageTier.CPU:
            data = loc.data
        elif loc.tier == StorageTier.SSD:
            data = torch.load(loc.path, map_location="cpu")
        else:
            return

        if self.gpu_count >= self.gpu_budget:
            self._evict_gpu_lru()

        gpu_data = data.to(self.device, non_blocking=True)
        event = torch.cuda.Event()
        event.record(self.prefetch_stream)
        self.prefetch_events[key] = event

        self.gpu_cache[key] = gpu_data
        self.gpu_count += 1
        self.locations[key].tier = StorageTier.GPU

    def wait_for_prefetch(self, layer_idx, expert_idx):
        """Block until prefetched expert is ready."""
        key = (layer_idx, expert_idx)
        if key in self.prefetch_events:
            self.prefetch_events[key].synchronize()
            del self.prefetch_events[key]
```

### History-Based Prefetching

A more sophisticated approach tracks expert usage patterns over recent tokens and predicts future usage:

```python
class HistoryBasedPredictor:
    """
    Predict which experts will be needed based on recent usage history.
    Uses a simple frequency model over a sliding window.
    """
    def __init__(self, num_layers, num_experts, window_size=32):
        self.num_layers = num_layers
        self.num_experts = num_experts
        self.window_size = window_size

        # Per-layer usage history: circular buffer of expert sets
        self.history = {
            layer: [] for layer in range(num_layers)
        }

    def record(self, layer_idx, expert_indices):
        """Record which experts were used at this layer."""
        h = self.history[layer_idx]
        h.append(set(expert_indices.tolist()))
        if len(h) > self.window_size:
            h.pop(0)

    def predict(self, layer_idx, top_k=4):
        """Predict which experts will be needed."""
        h = self.history[layer_idx]
        if not h:
            return list(range(min(top_k, self.num_experts)))

        # Count frequency of each expert in recent history
        freq = torch.zeros(self.num_experts)
        for expert_set in h:
            for e in expert_set:
                freq[e] += 1

        # Return top-k most frequent
        return freq.topk(top_k).indices.tolist()

    def get_prefetch_accuracy(self, layer_idx, actual_experts):
        """Measure how well predictions match actual usage."""
        predicted = set(self.predict(layer_idx, top_k=4))
        actual = set(actual_experts.tolist())
        if not actual:
            return 1.0
        return len(predicted & actual) / len(actual)
```

## 4. Quantization for Offloaded Experts

### Mixed-Precision Offloading

Keep active experts at higher precision (INT8 or FP16) on GPU, and store offloaded experts at INT4 on CPU/SSD to save memory and bandwidth:

```python
class QuantizedExpertStore:
    """
    Store offloaded experts in INT4 to reduce CPU memory and transfer bandwidth.
    Dequantize to FP16 on GPU after transfer.
    """
    def __init__(self, num_experts, expert_size_bytes, group_size=128):
        self.group_size = group_size
        self.experts_int4 = {}
        self.scales = {}
        self.zeros = {}

    def quantize_expert(self, layer_idx, expert_idx, fp16_params):
        """Quantize expert parameters to INT4."""
        key = (layer_idx, expert_idx)
        flat = fp16_params.float()

        # Group-wise quantization
        num_groups = flat.numel() // self.group_size
        grouped = flat[:num_groups * self.group_size].view(num_groups, self.group_size)

        # Compute scale and zero point per group
        mins = grouped.min(dim=1).values
        maxs = grouped.max(dim=1).values
        scales = (maxs - mins) / 15.0  # 4-bit: 0-15
        zeros = mins

        # Quantize
        quantized = ((grouped - zeros.unsqueeze(1)) / (scales.unsqueeze(1) + 1e-10))
        quantized = quantized.clamp(0, 15).to(torch.uint8)

        # Pack two INT4 values into one byte
        packed = quantized[:, ::2] | (quantized[:, 1::2] << 4)

        self.experts_int4[key] = packed
        self.scales[key] = scales.half()
        self.zeros[key] = zeros.half()

    def dequantize_expert(self, layer_idx, expert_idx, device="cuda:0"):
        """Dequantize INT4 expert to FP16 on target device."""
        key = (layer_idx, expert_idx)
        packed = self.experts_int4[key].to(device)
        scales = self.scales[key].to(device)
        zeros = self.zeros[key].to(device)

        # Unpack
        low = packed & 0x0F
        high = (packed >> 4) & 0x0F
        unpacked = torch.stack([low, high], dim=-1).view(
            scales.shape[0], self.group_size
        )

        # Dequantize
        dequantized = unpacked.float() * scales.unsqueeze(1).float() + zeros.unsqueeze(1).float()
        return dequantized.reshape(-1).half()

    def get_memory_savings(self, fp16_size_bytes):
        """Calculate memory savings from INT4 quantization."""
        int4_size = fp16_size_bytes / 4  # 4-bit vs 16-bit
        scale_overhead = fp16_size_bytes / (self.group_size * 2) * 2  # scales + zeros in FP16
        total_int4 = int4_size + scale_overhead
        return {
            "fp16_MB": fp16_size_bytes / 1e6,
            "int4_MB": total_int4 / 1e6,
            "ratio": fp16_size_bytes / total_int4,
        }
```

## 5. The Complete Offloading Pipeline

### Layer-by-Layer Execution

```python
class OffloadedMixtralInference:
    """
    Complete inference pipeline for Mixtral with expert offloading.
    """
    def __init__(
        self,
        model,
        gpu_experts=4,     # Keep 4 experts on GPU at once
        cpu_experts=48,    # Keep 48 experts in CPU RAM
        use_prefetch=True,
        use_quantization=True,
    ):
        self.model = model
        self.num_layers = 32
        self.num_experts = 8
        self.top_k = 2

        # Initialize offload manager
        self.manager = PrefetchingExpertManager(
            model=model,
            num_layers=self.num_layers,
            num_experts=self.num_experts,
            gpu_expert_budget=gpu_experts,
            cpu_expert_budget=cpu_experts,
        )

        # Initialize predictor
        self.predictor = HistoryBasedPredictor(
            self.num_layers, self.num_experts
        )

        # Keep attention layers and embeddings on GPU always
        self._pin_attention_to_gpu()

        # Move all experts to CPU initially
        self._offload_all_experts()

        self.use_prefetch = use_prefetch

    def _pin_attention_to_gpu(self):
        """Attention parameters stay on GPU permanently."""
        for layer in self.model.layers:
            layer.attention.to("cuda:0")
            layer.attention_norm.to("cuda:0")
            layer.ffn_norm.to("cuda:0")
            # Router stays on GPU (tiny: 4096 * 8 * 2 = 64KB)
            layer.moe.gate.to("cuda:0")

    def _offload_all_experts(self):
        """Move all expert parameters to CPU."""
        for layer_idx in range(self.num_layers):
            for expert_idx in range(self.num_experts):
                expert = self.model.layers[layer_idx].moe.experts[expert_idx]
                expert.to("cpu")

    def generate_token(self, hidden_states):
        """
        Forward pass for a single token through all layers.
        Manages expert loading/offloading per layer.
        """
        for layer_idx in range(self.num_layers):
            layer = self.model.layers[layer_idx]

            # 1. Attention (always on GPU)
            residual = hidden_states
            hidden_states = layer.attention_norm(hidden_states)
            hidden_states = layer.attention(hidden_states)
            hidden_states = residual + hidden_states

            # 2. MoE routing (router is on GPU)
            residual = hidden_states
            hidden_states = layer.ffn_norm(hidden_states)
            router_logits = layer.moe.gate(hidden_states)
            probs = torch.softmax(router_logits, dim=-1)
            top_k_probs, top_k_idx = probs.topk(self.top_k, dim=-1)
            weights = top_k_probs / top_k_probs.sum(dim=-1, keepdim=True)

            # 3. Prefetch next layer's experts
            if self.use_prefetch and layer_idx + 1 < self.num_layers:
                predicted = self.predictor.predict(layer_idx + 1, top_k=4)
                for e in predicted:
                    self.manager._async_load_to_gpu(layer_idx + 1, e)

            # 4. Load required experts for this layer
            expert_output = torch.zeros_like(hidden_states)
            unique_experts = top_k_idx.unique().tolist()

            for e in unique_experts:
                # Wait for prefetch if applicable
                self.manager.wait_for_prefetch(layer_idx, e)
                # Ensure expert is on GPU
                expert_module = self.manager.ensure_on_gpu(layer_idx, e)

                # Compute for all tokens routed to this expert
                for k in range(self.top_k):
                    mask = (top_k_idx[:, k] == e)
                    if mask.any():
                        out = expert_module(hidden_states[mask])
                        expert_output[mask] += weights[mask, k:k+1] * out

            # 5. Record usage for predictor
            self.predictor.record(layer_idx, top_k_idx.reshape(-1).unique())

            hidden_states = residual + expert_output

        return hidden_states
```

def benchmark_offloaded_inference(
    model,
    prompt_tokens=128,
    generation_tokens=256,
    batch_size=1,
    configs=None,

    """
    Benchmark different offloading configurations.
    """
    if configs is None:
        configs = [
            {"name": "Full GPU (2xA100 80GB)", "gpu_experts": 256, "prefetch": False},
            {"name": "Full GPU (1xA100 80GB, INT8)", "gpu_experts": 256, "prefetch": False},
            {"name": "Offload (4090 24GB, INT4, no prefetch)", "gpu_experts": 4, "prefetch": False},
            {"name": "Offload (4090 24GB, INT4, prefetch)", "gpu_experts": 4, "prefetch": True},
            {"name": "Offload (4090 24GB, INT4, aggressive prefetch)", "gpu_experts": 8, "prefetch": True},
        ]

    results = []
    for cfg in configs:
        engine = OffloadedMixtralInference(
            model,
            gpu_experts=cfg["gpu_experts"],
            use_prefetch=cfg["prefetch"],
        )

        # Warmup
        for _ in range(5):
            hidden = torch.randn(1, 1, 4096, device="cuda:0")
            engine.generate_token(hidden)

        # Benchmark
        torch.cuda.synchronize()
        import time
        start = time.perf_counter()
        for _ in range(generation_tokens):
            hidden = torch.randn(1, 1, 4096, device="cuda:0")
            engine.generate_token(hidden)
        torch.cuda.synchronize()
        elapsed = time.perf_counter() - start

        tokens_per_sec = generation_tokens / elapsed
        ms_per_token = elapsed * 1000 / generation_tokens

        results.append({
            "config": cfg["name"],
            "tokens_per_sec": tokens_per_sec,
            "ms_per_token": ms_per_token,
        })

    return results

## 7. SSD Offloading for Very Large Models

### When CPU RAM Is Insufficient

For larger MoE models (DeepSeek V3 with 256 experts at 671B parameters), even CPU RAM may not suffice. SSD offloading extends the hierarchy:

```python
import mmap
import os
import numpy as np

class SSDExpertStore:
    """
    Memory-mapped SSD expert store.
    Uses mmap for zero-copy access to expert parameters on SSD.
    """
    def __init__(self, cache_dir, num_layers, num_experts, expert_size_bytes):
        self.cache_dir = cache_dir
        self.expert_size = expert_size_bytes
        os.makedirs(cache_dir, exist_ok=True)

        # Create a single large file for all experts
        total_size = num_layers * num_experts * expert_size_bytes
        self.data_path = os.path.join(cache_dir, "experts.bin")

        if not os.path.exists(self.data_path):
            # Initialize file
            with open(self.data_path, 'wb') as f:
                f.seek(total_size - 1)
                f.write(b'\0')

        # Memory-map the file
        self.file = open(self.data_path, 'r+b')
        self.mmap = mmap.mmap(self.file.fileno(), 0)

        self.num_layers = num_layers
        self.num_experts = num_experts

    def _offset(self, layer_idx, expert_idx):
        return (layer_idx * self.num_experts + expert_idx) * self.expert_size

    def write_expert(self, layer_idx, expert_idx, data_bytes):
        offset = self._offset(layer_idx, expert_idx)
        self.mmap[offset:offset + len(data_bytes)] = data_bytes

    def read_expert(self, layer_idx, expert_idx):
        offset = self._offset(layer_idx, expert_idx)
        data = self.mmap[offset:offset + self.expert_size]
        # Convert to tensor
        arr = np.frombuffer(data, dtype=np.float16)
        return torch.from_numpy(arr.copy())

    def read_expert_async(self, layer_idx, expert_idx, callback):
        """Read expert on a background thread."""
        def _read():
            tensor = self.read_expert(layer_idx, expert_idx)
            callback(layer_idx, expert_idx, tensor)

        thread = threading.Thread(target=_read)
        thread.start()
        return thread
```

## 8. Practical Implementation: llama.cpp and ExLlamaV2

### llama.cpp MoE Offloading

llama.cpp supports partial GPU offloading of MoE models via its `--n-gpu-layers` flag. The implementation:

1. Layers are assigned to GPU sequentially. If you specify `--n-gpu-layers 16`, the first 16 transformer layers (including all 8 experts per layer) go to GPU.
2. Remaining layers stay on CPU.
3. Expert offloading within a layer is not supported — it is all-or-nothing per layer.

## 9. Expert Caching Strategies

### LRU vs Frequency-Based Eviction

```python
class FrequencyAwareCache:
    """
    Expert cache that considers both recency and frequency.
    Evicts the expert with the lowest score = alpha * frequency + (1-alpha) * recency.
    """
    def __init__(self, capacity, alpha=0.3, decay=0.99):
        self.capacity = capacity
        self.alpha = alpha
        self.decay = decay
        self.cache = {}  # key -> (data, score)
        self.access_count = {}
        self.step = 0

    def access(self, key):
        self.step += 1
        self.access_count[key] = self.access_count.get(key, 0) + 1

        if key in self.cache:
            self._update_score(key)
            return self.cache[key][0]
        return None

    def insert(self, key, data):
        if len(self.cache) >= self.capacity:
            self._evict()
        self.cache[key] = (data, self._compute_score(key))

    def _compute_score(self, key):
        freq = self.access_count.get(key, 0)
        recency = 1.0 / (self.step - self.access_count.get(f"{key}_last_step", 0) + 1)
        self.access_count[f"{key}_last_step"] = self.step
        return self.alpha * freq + (1 - self.alpha) * recency

    def _update_score(self, key):
        data = self.cache[key][0]
        self.cache[key] = (data, self._compute_score(key))

    def _evict(self):
        # Decay all scores
        for key in self.cache:
            data, score = self.cache[key]
            self.cache[key] = (data, score * self.decay)

        # Evict lowest score
        min_key = min(self.cache.keys(), key=lambda k: self.cache[k][1])
        del self.cache[min_key]
```

### Expert Popularity Distribution

In practice, expert popularity follows a power law: a few experts are accessed very frequently, and many are accessed rarely.

```python
def measure_expert_popularity(model, dataset, tokenizer, num_layers=32, num_experts=8):
    """Measure the popularity distribution of experts across a dataset."""
    counts = torch.zeros(num_layers, num_experts)

    for text in dataset:
        tokens = tokenizer.encode(text, return_tensors="pt").to("cuda")
        with torch.no_grad():
            outputs = model(tokens, output_router_logits=True)

        for layer_idx, logits in enumerate(outputs.router_logits):
            top2 = logits.squeeze(0).topk(2, dim=-1).indices
            for e in top2.reshape(-1).tolist():
                counts[layer_idx, e] += 1

    # Normalize per layer
    popularity = counts / counts.sum(dim=1, keepdim=True)
    return popularity
```

## 10. Advanced: Speculative Expert Loading

### Pipeline Expert Prefetching

The most aggressive optimization: run the router for all layers before running any expert computation, then load all needed experts in one batch transfer:

```python
class SpeculativeExpertLoader:
    """
    Run all routers first to get the complete expert schedule,
    then batch-load all needed experts in a single PCIe transfer.
    """
    def __init__(self, model, device="cuda:0"):
        self.model = model
        self.device = device

    def get_expert_schedule(self, hidden_states):
        """
        Run through all layers' routers to get the complete
        expert assignment for this token.
        Returns: list of (layer_idx, expert_indices) tuples.
        """
        schedule = []
        x = hidden_states

        for layer_idx, layer in enumerate(self.model.layers):
            # Run attention (already on GPU)
            residual = x
            x = layer.attention_norm(x)
            x = layer.attention(x)
            x = residual + x

            # Run router only (no expert computation)
            normed = layer.ffn_norm(x)
            logits = layer.moe.gate(normed)
            top_k_idx = logits.topk(2, dim=-1).indices

            schedule.append((layer_idx, top_k_idx.unique().tolist()))

            # Problem: x depends on expert output for accurate routing
            # at subsequent layers. So this is speculative.
            # Use x without expert contribution (residual only)

        return schedule

    def batch_load_experts(self, schedule, manager):
        """Load all needed experts in one batch."""
        needed = set()
        for layer_idx, experts in schedule:
            for e in experts:
                needed.add((layer_idx, e))

        # Sort by layer for sequential access
        needed = sorted(needed)

        # Batch transfer
        for layer_idx, expert_idx in needed:
            manager.ensure_on_gpu(layer_idx, expert_idx)
```

<Callout type="warning" title="Speculative Loading Accuracy">
Speculative expert loading is imprecise because later layers' routing depends on earlier layers' expert outputs (which we skip in the speculation pass). In practice, speculative accuracy degrades for later layers — approximately 90% accurate for layer $N+1$ but only 60% accurate for layer $N+16$. A hybrid approach that re-speculates every 4 layers works better: speculate layers $N$ to $N+3$, execute them, then speculate $N+4$ to $N+7$.
</Callout>

## 11. Memory Budget Planning

```python
def plan_offloading_config(
    total_params_B,
    expert_fraction,
    num_experts,
    num_layers,
    gpu_memory_gb,
    cpu_memory_gb,
    precision="int4",
):
    """
    Plan the optimal offloading configuration given hardware constraints.
    """
    bytes_per_param = {
        "fp16": 2, "int8": 1, "int4": 0.5
    }[precision]

    # Memory breakdown
    expert_params = total_params_B * 1e9 * expert_fraction
    non_expert_params = total_params_B * 1e9 * (1 - expert_fraction)

    expert_memory_gb = expert_params * bytes_per_param / 1e9
    non_expert_memory_gb = non_expert_params * bytes_per_param / 1e9

    # Non-expert params always on GPU
    gpu_available = gpu_memory_gb - non_expert_memory_gb

    # Reserve 2GB for KV cache and activations
    gpu_available -= 2.0

    # How many experts fit on GPU?
    expert_size_gb = expert_memory_gb / (num_experts * num_layers)
    gpu_experts = int(gpu_available / expert_size_gb)

    # How many on CPU?
    remaining = num_experts * num_layers - gpu_experts
    cpu_available_gb = cpu_memory_gb - 4.0  # Reserve 4GB for OS
    cpu_experts = min(remaining, int(cpu_available_gb / expert_size_gb))

    # Rest on SSD
    ssd_experts = remaining - cpu_experts

    return {
        "expert_size_mb": expert_size_gb * 1000,
        "gpu_experts": gpu_experts,
        "cpu_experts": cpu_experts,
        "ssd_experts": ssd_experts,
        "gpu_memory_used_gb": non_expert_memory_gb + gpu_experts * expert_size_gb + 2.0,
        "cpu_memory_used_gb": cpu_experts * expert_size_gb + 4.0,
    }
```

## 12. Summary

Expert offloading makes MoE models accessible on consumer hardware at the cost of throughput. The key techniques:

1. **Tiered storage**: GPU VRAM for active experts, CPU RAM for warm experts, SSD for cold storage.
2. **Prefetching**: Use router logits and usage history to predict and preload next-layer experts, hiding 70-90% of transfer latency.
3. **Quantized offloading**: Store offloaded experts at INT4 to reduce both memory and bandwidth requirements.
4. **Smart caching**: Frequency-aware eviction outperforms simple LRU because expert popularity follows a power law.

With aggressive prefetching and INT4 quantization, Mixtral 8x7B runs at 22 tokens/sec on a single RTX 4090 — usable for interactive applications. Without prefetching, it drops to 8 tokens/sec. The gap between full-GPU and offloaded inference is narrowing as PCIe bandwidth increases and prefetching algorithms improve.
